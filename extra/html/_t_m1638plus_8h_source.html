60 // Send ASCII value to seven segment, pass position 0-7 and ASCII value byte

63 // Same as displayASCII function but turns on dot/decimal point as well

74 // Display an integer and leading zeros optional
75 // Param 1 :: integer to display 2^32
76 // Param 2 :: bool leading zeros , true on , false off
77 // Param 3 :: enum text alignment , left or right alignment
78 void displayIntNum(unsigned long number, boolean leadingZeros = true, AlignTextType_e = TMAlignTextLeft);

80 // Divides the display into two nibbles and displays a Decimal number in each.
81 // takes in two numbers 0-9999 for each nibble ,
82 // and leading zeros optional
83 // Param 1 :: upper nibble integer 2^16
84 // Param 2 :: lower nibble integer 2^16
85 // Param 3 :: bool leading zeros , true on , false off
86 // Param 4 :: enum text alignment , left or right alignment
87 void DisplayDecNumNibble(uint16_t numberUpper, uint16_t numberLower, boolean leadingZeros = true, AlignTextType_e = TMAlignTextLeft);
88
89 // Set the LEDs. passed one 16bit integer.
90 // MODEL 3:
91 //MSB byte for the green LEDs, LS byte for the red LEDs (0xgreenred)
92 //ie. 0xE007 1110 0000 0000 0111 results in L8-L0 GGGX XRRR, NOTE L8 is RHS on display
93 // MODEL 1:
94 // MSB byte 1 for red LED , LSB byte n/a set to 0x00 (0xleds, 0xXX)
95 //i.e 0xF100 1111 0000 L8-L0 RRRRXXX0 NOTE L8 is RHS on display
96 void setLEDs(uint16_t greenred);
97
98 // Set an LED, pass it LED position 0-7 and value 0 or 1 , L1-L8
