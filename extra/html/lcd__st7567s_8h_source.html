30 // LCD is driven by ST7567A IC.
31 // Define instructions on page 37 of the ST7567A data sheet.
32 enum ST7567A_C {
33 displayON = 0xaf, //display on.
34 displayOFF = 0xae, //display off.
35
36 startLine = 0x40, //Set display start line.
37 restart = 0xe2, //restart command.
38 seg = 0xa0, //Set scan direction of SEG, SEG = 1-0-1-0-0-0-0-MX;
39 //MX=1-->reverse direction, MX=0-->normal direction.
40
41 INV_normal = 0xa6, //normal display.
42 AP_normal = 0xa4, //normal display.
43
44 BS = 0xa3, //select bias setting, bias = 1-0-1-0-0-0-1-BS, BS(0)=1/9, BS(1)=1/7(at 1/65 duty).
45 com = 0xc8, //Set output direction of COM, COM = 1-1-0-0-MY-X-X-X;
46 //MY=1-->reverse direction, MY=0-->normal direction.
47
48 powerCon1 = 0x2c, //Controls the built-in power circuits;
49 powerCon2 = 0x2e, //Define instructions on page 43 of the ST7567A data sheet.
50 powerCon3 = 0x2f,
51
52 /********Adjust display brightness********/
53 regRatio = 0x22, //Controls the regulation ratio of the built-in regulator;
54 //Define instructions on page 43 of the ST7567A data sheet.
55 EV1 = 0x81, //Define instructions on page 44 of the ST7567A data sheet.
56 EV2 = 0x30,
57
58 /**********entension command set**********/
59 exit_EC = 0xfe, //exit extension command set.
60 enter_EC = 0xff, //enter extension command set.
61
62 DSM_ON = 0x72, //display setting mode on.
63 DSM_OFF = 0x70, //display setting mode off.
64 DT = 0xd6, //set the display duty, DT=33.
65 BA = 0x90, //BA=1/9, selects LCD bias ratio for the internal voltage follower to drive the LCD.
66 //This command has priority over the Bias Select (BS).
67 FR = 0x9d, //specifies the frame rate for duty.
68
69 RMW = 0xe0, //Read-modify-Write command on page 42 of ST7567A datasheet.
70 END = 0xee,
71 };
