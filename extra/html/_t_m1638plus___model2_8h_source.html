44 // Sets the brightness level on a scale of brightness = 0 to 7.

50 // Function Desc:Read push buttons function
51 // User may have to deb-ounce buttons depending on application.
52 // See [URL LINK](https://github.com/gavinlyonsrepo/Arduino_Clock_3)
53 // For de-bounce practical example(model 1)
54 // model 2 example here in ADC file TM1638plus_ADC_TEST_Model2.ino.
55 // Note there are two different read button functions See Also ReadKeyTwo

57 // Returns: a byte with decimal value of buttons 1-16 (1 for S1 2 for S2 ... 16 for S16 ).
58 uint8_t ReadKey16(void);

60 // Function Desc: Reads the push buttons: alternate function to ReadKeys16
61 // Can detect multiple key presses , See notes section in readme regarding
62 // Problems with seven segment display when pressing certain keys in combination
63 // Returns: A 16 bit integer where each bit correpsond to a switch
64 // Eg:
65 // S1 = 0x0001
66 // S16 = 0x8000
67 // S16 + S1 together = 0x8001
68 uint16_t ReadKey16Two(void);
69
70 // Send seven segment value module
71 // pass segment 0-7, byte of data corresponding to segments abcdefg(dp) 01234567.
72 // a is 0 , dp is 7 , segmentValue is which segments are off or on for each digit. d7d6d5d54d3d2d1
73 // Have to call it 8 times in row to set all segments.
74 // to set all "a" on send (0x00,0xFF)
75 // to set all segment "g" off (0x06,0X00)
76 void DisplaySegments(uint8_t segment, uint8_t segmentValue);
77
78 // Display a Hexadecimal number ,
79 // Param 1 :: upper nibble integer 2^16
80 // Param 2 :: lower nibble integer 2^16
81 // Param 3 :: byte dots turn on decimal points
82 // Param 4 :: bool leading zeros , true on , false off
83 // Param 5 :: enum text alignment , left or right alignment
84 // converts to string internally
85 void DisplayHexNum(uint16_t numberUpper, uint16_t numberLower, byte dots, boolean leadingZeros = true, AlignTextType_e = TMAlignTextLeft);
86
87 // Display a decimal number ,
88 // Param 1 :: 2^32 unsigned long
89 // Param 2 :: byte for decimal point display switch's on decimal point for those positions
90 // Param 3 :: leading zeros , true on , false off
91 // Param 4 :: enum text alignment , left or right
92 // converts to string internally
93 void DisplayDecNum(unsigned long number, byte dots, boolean leadingZeros = true, AlignTextType_e = TMAlignTextLeft);
94
95 // Display a string , takes a string and byte for decimal point display
96 // Takes in string , converts it to ASCII using the font and masks for the decimal point
97 // Then passes array of eight ASCII bytes to DisplayValues function
98 void DisplayStr(const char* string, const word dots = 0);
99
100 // Takes in Array of 8 ASCII bytes , Called from DisplayStr
101 // Scans each ASCII byte converts to array of 8 segment bytes where each byte represents a segment
102 // i.e. byte 0 represents a in segment and then each bit represents the a segment in each digit.
103 // So for "00000005" is converted by DisplayStr to hex "3F 3F 3F 3F 3F 3F 3F 6D" where left is first digit.
104 // this is converted to hex "FF FE FF FF FE FF 01 00" Where left is first segment
105 // So "a" segment is turned on for all bytes and "b" is on for all except last digit.
106 // The bits are mapping below abcdefg(dp) = 01234567 !
107 // see for mapping of seven segment to digit https://en.wikipedia.org/wiki/Seven-segment_display
108 void ASCIItoSegment(const byte values[]);
109
110 //Divides the display into two nibbles and displays a Decimal number in each.
